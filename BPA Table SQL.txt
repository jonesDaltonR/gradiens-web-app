CREATE TABLE [dbo].[USER]
(
	[USER_ID] INT NOT NULL PRIMARY KEY,
	[USER_USERNAME] VARCHAR(64) NOT NULL, 
    [USER_PASSWORD] VARCHAR(64) NOT NULL, 
    [USER_FIRST_NAME] VARCHAR(64) NOT NULL, 
    [USER_MIDDLE_NAME] VARCHAR(64) NULL, 
    [USER_LAST_NAME] VARCHAR(64) NOT NULL, 
    [USER_DATE_OF_BIRTH] DATE NOT NULL, 
    [USER_GENDER] VARCHAR(6) NOT NULL, 
    [USER_EMAIL] VARCHAR(128) NULL
)

CREATE TABLE [dbo].[ADMIN]
(
	[ADMIN_ID] INT NOT NULL PRIMARY KEY, 
    [ADMIN_CONTROLS] BIT NOT NULL DEFAULT 0
)

CREATE TABLE [dbo].[REG_USER]
(
	[REG_USER_ID] INT NOT NULL PRIMARY KEY, 
    [REG_USER_PAYMENT_METHOD] VARCHAR(64) NOT NULL, 
    [REG_USER_HOME_AIRPORT] VARCHAR(64) NOT NULL, 
    [REG_USER_SEATING_PREFERENCE] VARCHAR(32) NOT NULL, 
    [REG_USER_LOCATION_ID] INT NOT NULL, 
)

CREATE TABLE [dbo].[LOCATION]
(
	[LOCATION_ID] INT NOT NULL PRIMARY KEY, 
    [LOCATION_CITY_NAME] VARCHAR(64) NOT NULL, 
    [LOCATION_COUNTRY_NAME] VARCHAR(64) NOT NULL, 
    [LOCATION_ADDRESS] VARCHAR(64) NOT NULL, 
    [LOCATION_LONGITUDE] VARCHAR(16) NOT NULL,
    [LOCATION_LATITUDE] VARCHAR(16) NOT NULL
)

CREATE TABLE [dbo].[TRAVEL_RES]
(
	[TRAVEL_RES_ID] INT NOT NULL PRIMARY KEY, 
    [RU_ID] INT NOT NULL, 
    [SEATING_ID] INT NOT NULL, 
    [TRAVEL_RES_TOTAL] DECIMAL(10, 2) NOT NULL, 
    [TRAVEL_RES_START_DATE] DATE NOT NULL, 
    [TRAVEL_RES_END_DATE] DATE NOT NULL, 
    [TRAVEL_RES_ACTIVE] BIT NOT NULL DEFAULT 1
)

CREATE TABLE [dbo].[HOTEL]
(
	[HOTEL_ID] INT NOT NULL PRIMARY KEY, 
    [HOTEL_LOCATION_ID] INT NOT NULL, 
    [HOTEL_NAME] VARCHAR(64) NOT NULL, 
    [HOTEL_COMPANY] VARCHAR(64) NOT NULL, 
    [HOTEL_ROOM_NUMBER] INT NOT NULL, 
    [HOTEL_CHECK_IN_DATE] DATE NOT NULL, 
    [HOTEL_CHECK_OUT_DATE] DATE NOT NULL, 
    [COST] DECIMAL(10, 2) NOT NULL
)

CREATE TABLE [dbo].[SEATING_RES]
(
	[SEATING_RES_ID] INT NOT NULL PRIMARY KEY, 
    [SEATING_RES_SEAT] VARCHAR(4) NOT NULL, 
    [SEATING_RES_COST] DECIMAL(10, 2) NOT NULL, 
    [SEATING_RES_CAPACITY] INT NOT NULL, 
    [SEATING_RES_ARRIVE_DATE] DATE NOT NULL, 
    [SEATING_RES_DEPART_DATE] DATE NOT NULL
)

CREATE TABLE [dbo].[TRAIN_STATION]
(
	[STATION_ID] INT NOT NULL PRIMARY KEY, 
    [STATION_NAME] VARCHAR(64) NOT NULL, 
    [STATION_LOCATION] INT NOT NULL
)

CREATE TABLE [dbo].[TRAIN]
(
	[TRAIN_ID] INT NOT NULL PRIMARY KEY, 
    [TRAIN_COMPANY] VARCHAR(64) NOT NULL, 
    [STATION_ID] INT NOT NULL, 
    [SEATING_ID] INT NOT NULL
)

CREATE TABLE [dbo].[AIRPORT]
(
	[AIRPORT_ID] INT NOT NULL PRIMARY KEY, 
    [AIRPORT_LOCATION] INT NOT NULL, 
    [AIRPORT_NAME] VARCHAR(64) NOT NULL
)

CREATE TABLE [dbo].[AIRPLANE]
(
	[AIRPLANE_ID] INT NOT NULL PRIMARY KEY, 
    [AIRPLANE_AIRLINE_COMPANY] NCHAR(10) NOT NULL, 
    [AIRPORT_ID] INT NOT NULL, 
    [SEATING_ID] INT NOT NULL
)

CREATE TABLE [dbo].[BUS_STOP]
(
	[BUS_STOP_ID] INT NOT NULL PRIMARY KEY, 
    [BUS_STOP_NAME] VARCHAR(64) NOT NULL, 
    [BUS_STOP_LOCATION] INT NOT NULL
)

CREATE TABLE [dbo].[BUS]
(
	[BUS_ID] INT NOT NULL PRIMARY KEY, 
    [BUS_COMPANY] VARCHAR(64) NOT NULL, 
    [BUS_STOP_ID] INT NOT NULL, 
    [SEATING_ID] NCHAR(10) NOT NULL
)

CREATE TABLE [dbo].[HARBOR]
(
	[HARBOR_ID] INT NOT NULL PRIMARY KEY, 
    [HARBOR_NAME] VARCHAR(64) NOT NULL, 
    [HARBOR_LOCATION] INT NOT NULL
)

CREATE TABLE [dbo].[CRUISE]
(
	[CRUISE_ID] INT NOT NULL PRIMARY KEY, 
    [CRUISE_NAME] VARCHAR(64) NOT NULL, 
    [CRUISE_COMPANY] VARCHAR(64) NOT NULL,
